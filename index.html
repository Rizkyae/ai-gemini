JavaScript Example 
const response = await fetch('https://api.cosmochat.ai/v1/chat', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${API_KEY}`
    },
    body: JSON.stringify({
        model: 'cosmo-7b',
        messages: [
            {
                role: 'user',
                content: 'Hello, how are you?'
            }
        ],
        temperature: 0.7
    })
});

const data = await response.json();
